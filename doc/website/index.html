fn sigma_fn(n: int) -> int {
    let mut s = 0
    for d in 1..=n {
        if n % d == 0 { s = s + d }
    }
    return s
}

fn phi_fn(n: int) -> int {
    let mut count = 0
    for k in 1..=n {
        if gcd(k, n) == 1 { count = count + 1 }
    }
    return count
}

proof n6_uniqueness {
    for n in 2..101 {
        if sigma_fn(n) * phi_fn(n) == n * tau(n) {
            assert n == 6   // only n=6 satisfies the identity
        }
    }
}
